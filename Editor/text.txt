struct UVs
{
    float u, v, width, height;

    UVs()
        : u(0.f), v(0.f), width(0.f), height(0.f)
    {
    }
};

class Camera2D
{
public:
    Camera2D() : Camera2D(640, 480) {}

    Camera2D(int width, int height)
        : m_Width(width),
          m_Height(height),
          m_Scale(1.0f),
          m_Pos(0.0f, 0.0f),
          m_CameraMatrix(1.0f),
          m_Projection(1.0f),
          m_NeedsUpdate(true)
    {
        m_Projection = ortho(0.0f, (float)m_Width, (float)m_Height, 0.0f);
        //m_Projection = ortho(0.0f, (float)m_Width, 0, (float)m_Height);
    }

    void SetPosition(const vec2& pos)
    {
        m_Pos = pos;
        m_NeedsUpdate = true;
    }

    void SetScale(float scaleValue)
    {
        m_Scale = scaleValue;
        m_NeedsUpdate = true;
    }

    void Update()
    {
        if (!m_NeedsUpdate)
            return;

        mat4 translation = translate(vec3(-m_Pos.x, -m_Pos.y, 0.0f));
        mat4 scaling = scale(vec3(m_Scale, m_Scale, 1.0f));

        m_CameraMatrix = m_Projection * translation * scaling;
        m_NeedsUpdate = false;
    }

    const mat4& GetCameraMatrix() const
    {
        return m_CameraMatrix;
    }

private:
    int m_Width;
    int m_Height;
    float m_Scale;
    vec2 m_Pos;

    mat4 m_CameraMatrix;
    mat4 m_Projection;
    bool m_NeedsUpdate;
};

int main(int argc, const char* argv[])
{
    Window window;

    if (!gladLoadGLLoader((GLADloadproc)glfwGetProcAddress))
    {
        std::cout << "glad failed init\n";
        return -1;
    }

    auto tex = FrostEngine::TextureLoader::Create(FrostEngine::Texture::TextureType::BLENDED, "../../../data/images/awesomeface.png");
    
    auto shader = FrostEngine::ShaderLoader::Create("../../../Engine/Assets/Shaders/vertex.glsl", "../../../Engine/Assets/Shaders/fragment.glsl");

    

    UVs uvs{};

    auto generateUVs = [&](float startX, float startY, float spriteWidth, float spriteHeight)
    {
        uvs.u = startX / tex->GetWidth();
        uvs.v = startY / tex->GetHeight();
        uvs.width = spriteWidth / tex->GetWidth();
        uvs.height = spriteHeight / tex->GetHeight();
    };

    generateUVs(0.0f, 0.0f, (float)tex->GetWidth(), (float)tex->GetHeight());

    float x = 125.0f;
    float y = 100.0f;
    float w = 100.0f;
    float h = 100.0f;

    float vertices[] =
    {
        // position      // uv
        x,     y,       uvs.u,             uvs.v,
        x + w, y,       uvs.u + uvs.width, uvs.v,
        x + w, y + h,   uvs.u + uvs.width, uvs.v + uvs.height,
        x,     y + h,   uvs.u,             uvs.v + uvs.height
    };

    unsigned int indices[] =
    {
        0, 1, 2,
        2, 3, 0
    };

    float xx = 150.0f;
    float yy = 150.0f;
    float ww = 150.0f;
    float hh = 150.0f;

    float verticess[] =
    {
        // position      // uv
        xx,     yy,       uvs.u,             uvs.v,
        xx + ww, yy,       uvs.u + uvs.width, uvs.v,
        xx + ww, yy + hh,   uvs.u + uvs.width, uvs.v + uvs.height,
        xx,     yy + hh,   uvs.u,             uvs.v + uvs.height
    };

    unsigned int indicess[] =
    {
        0, 1, 2,
        2, 3, 0
    };

    Camera2D camera(window.Width(), window.Height());   

    unsigned int vao, vbo, ebo;
    unsigned int vaoo, vboo, eboo;
    glGenVertexArrays(1, &vao);
    glGenBuffers(1, &vbo);
    glGenBuffers(1, &ebo);

    glBindVertexArray(vao);

    glBindBuffer(GL_ARRAY_BUFFER, vbo);
    glBufferData(GL_ARRAY_BUFFER, sizeof(vertices), vertices, GL_STATIC_DRAW);

    glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, ebo);
    glBufferData(GL_ELEMENT_ARRAY_BUFFER, sizeof(indices), indices, GL_STATIC_DRAW);

    glVertexAttribPointer(0, 2, GL_FLOAT, GL_FALSE, sizeof(float) * 4, (void*)0);
    glEnableVertexAttribArray(0);

    glVertexAttribPointer(1, 2, GL_FLOAT, GL_FALSE, sizeof(float) * 4, (void*)(sizeof(float) * 2));
    glEnableVertexAttribArray(1);

    glBindBuffer(GL_ARRAY_BUFFER, 0);
    glBindVertexArray(0);

    glGenVertexArrays(1, &vaoo);
    glGenBuffers(1, &vboo);
    glGenBuffers(1, &eboo);

    glBindVertexArray(vaoo);

    glBindBuffer(GL_ARRAY_BUFFER, vboo);
    glBufferData(GL_ARRAY_BUFFER, sizeof(verticess), verticess, GL_STATIC_DRAW);

    glBindBuffer(GL_ELEMENT_ARRAY_BUFFER, eboo);
    glBufferData(GL_ELEMENT_ARRAY_BUFFER, sizeof(indicess), indicess, GL_STATIC_DRAW);

    glVertexAttribPointer(0, 2, GL_FLOAT, GL_FALSE, sizeof(float) * 4, (void*)0);
    glEnableVertexAttribArray(0);

    glVertexAttribPointer(1, 2, GL_FLOAT, GL_FALSE, sizeof(float) * 4, (void*)(sizeof(float) * 2));
    glEnableVertexAttribArray(1);

    glBindBuffer(GL_ARRAY_BUFFER, 0);
    glBindVertexArray(0);

    Test::hello();
    Test::addInt(2);
    Test::Get().addFloat(2.f);
    //Test::Get().hellov2();
    while (window.IsRunning())
{
    glfwPollEvents();

    if (glfwWindowShouldClose(window.Handle()))
    {
        window.IsRunning() = false;
    }

    glViewport(0, 0, window.Width(), window.Height());
    glClearColor(1.0f, 0.2f, 0.5f, 1.0f);
    glClear(GL_COLOR_BUFFER_BIT | GL_STENCIL_BUFFER_BIT);

    glEnable(GL_STENCIL_TEST);

    // ---------------------------------
    // 1. Draw stencil mask (window quad)
    // ---------------------------------
    glStencilMask(0xFF);                         // enable writing to stencil
    glStencilFunc(GL_ALWAYS, 1, 0xFF);          // every fragment passes
    glStencilOp(GL_KEEP, GL_KEEP, GL_REPLACE);  // write 1 where fragments pass

    glColorMask(GL_FALSE, GL_FALSE, GL_FALSE, GL_FALSE); // don't write colors

    shader->Bind();
    shader->SetMatrix4x4("proj", camera.GetCameraMatrix());

    glBindVertexArray(vaoo);
    glDrawElements(GL_TRIANGLES, 6, GL_UNSIGNED_INT, nullptr);

    // ---------------------------------
    // 2. Draw real object only where stencil == 1
    // ---------------------------------
    glColorMask(GL_TRUE, GL_TRUE, GL_TRUE, GL_TRUE); // restore color writes

    glStencilMask(0x00);                    // do not modify stencil anymore
    glStencilFunc(GL_NOTEQUAL, 1, 0xFF);      // only draw where stencil is 1
    glStencilOp(GL_KEEP, GL_KEEP, GL_KEEP);

    camera.Update();
    shader->Bind();
    shader->SetMatrix4x4("proj", camera.GetCameraMatrix());

    glActiveTexture(GL_TEXTURE0);
    glBindTexture(GL_TEXTURE_2D, tex->GetID());

    glBindVertexArray(vao);
    glDrawElements(GL_TRIANGLES, 6, GL_UNSIGNED_INT, nullptr);

    glBindVertexArray(0);
    glDisable(GL_STENCIL_TEST);

    glfwSwapBuffers(window.Handle());
}

    glDeleteVertexArrays(1, &vao);
    glDeleteBuffers(1, &vbo);
    glDeleteBuffers(1, &ebo);
    //glDeleteTextures(1, &tex->GetID());

    return 0;
}
